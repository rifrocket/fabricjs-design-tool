React Integration

Before After 
import React, { useRef, useEffect } from 'react';
import { fabric } from 'fabric';

function MyCanvas() {
  const canvasRef = useRef(null);
  
  useEffect(() => {
    const canvas = new fabric.Canvas(canvasRef.current, {
      width: 800,
      height: 600
    });
    
    return () => canvas.dispose();
  }, []);
  
  return <canvas ref={canvasRef} />;
}
 
import React from 'react';
import { CanvasWrapper } from 'fabricjs-design-tool/ui';

function MyCanvas() {
  return (
    <CanvasWrapper
      width={800}
      height={600}
      onCanvasReady={(canvas) => {
        // Canvas is ready to use
        console.log('Canvas ready:', canvas);
      }}
    />
  );
}

Event Handling

Before After 
// Manual event handling
canvas.on('selection:created', (e) => {
  console.log('Selected:', e.selected);
});

canvas.on('object:modified', (e) => {
  console.log('Modified:', e.target);
});

canvas.on('mouse:wheel', (opt) => {
  const delta = opt.e.deltaY;
  let zoom = canvas.getZoom();
  zoom *= 0.999 ** delta;
  if (zoom > 20) zoom = 20;
  if (zoom < 0.01) zoom = 0.01;
  canvas.setZoom(zoom);
  opt.e.preventDefault();
  opt.e.stopPropagation();
});
 
// Built-in event handling with hooks
import { useCanvasManager } from 'fabricjs-design-tool';

const canvasManager = useCanvasManager(canvas, {
  onSelectionChange: (selected) => {
    console.log('Selected:', selected);
  },
  onObjectModified: (object) => {
    console.log('Modified:', object);
  },
  enableZoom: true, // Auto-handles mouse wheel zoom
  zoomToPoint: true
});

⚠️Breaking Changes

  • Import paths: Some imports have changed from fabric to fabricjs-design-tool
  • Event signatures: Some event callbacks have slightly different parameters
  • Configuration: Some canvas options are now handled by the canvas manager
  • Dependencies: React components require React 16.8+ for hooks support

💡Migration Tips

  • Gradual migration: You can migrate piece by piece, starting with core functionality
  • Coexistence: The library can coexist with existing Fabric.js code during transition
  • Testing: Thoroughly test interactive features after migration
  • Performance: Monitor performance improvements with built-in optimizations
  • Documentation: Refer to examples for complex migration scenarios
