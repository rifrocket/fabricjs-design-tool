CanvasWrapper Props

interface CanvasWrapperProps {
  width?: number;           // Canvas width (default: 800)
  height?: number;          // Canvas height (default: 600)
  backgroundColor?: string; // Canvas background (default: '#ffffff')
  className?: string;       // CSS class
  style?: React.CSSProperties;
  onCanvasReady?: (canvas: fabric.Canvas) => void;
  onSelectionChange?: (selected: fabric.Object[]) => void;
  onObjectModified?: (object: fabric.Object) => void;
}

Complete Example

import React, { useState } from 'react';
import { CanvasWrapper } from 'fabricjs-design-tool/ui';

function MyApp() {
  const [canvas, setCanvas] = useState(null);
  const [selected, setSelected] = useState(null);
  
  return (
    <div style={{ height: '100vh', display: 'flex', flexDirection: 'column' }}>
      <CanvasWrapper
        width={1000}
        height={700}
        backgroundColor="#f8f9fa"
        onCanvasReady={setCanvas}
        onSelectionChange={(objects) => setSelected(objects[0])}
        onObjectModified={(obj) => console.log('Modified:', obj)}
      />
    </div>
  );
}

üé®Complete Layout Example

import React, { useState } from 'react';
import { 
  CanvasWrapper,
  Header,
  LeftSidebar,
  RightSidebar,
  BottomToolbar,
  QRCodeDialog,
  ShapeDialog,
  KeyboardShortcutsModal
} from 'fabricjs-design-tool/ui';

function DesignApp() {
  const [canvas, setCanvas] = useState(null);
  const [selectedObject, setSelectedObject] = useState(null);
  const [zoom, setZoom] = useState(1);
  const [showQR, setShowQR] = useState(false);
  const [showShapes, setShowShapes] = useState(false);
  const [showHelp, setShowHelp] = useState(false);

  return (
    <div className="design-app">
      <Header 
        onExport={() => console.log('Export')}
        onShowQR={() => setShowQR(true)}
        onShowShapes={() => setShowShapes(true)}
        onShowHelp={() => setShowHelp(true)}
      />
      
      <div className="design-body">
        <LeftSidebar 
          canvas={canvas}
          onToolSelect={(tool) => console.log('Tool:', tool)}
        />
        
        <CanvasWrapper
          onCanvasReady={setCanvas}
          onSelectionChange={(objs) => setSelectedObject(objs[0])}
          width={1200}
          height={800}
        />
        
        <RightSidebar 
          selectedObject={selectedObject}
          onPropertyChange={(prop, value) => {
            if (selectedObject) {
              selectedObject.set(prop, value);
              canvas?.renderAll();
            }
          }}
        />
      </div>
      
      <BottomToolbar 
        zoom={zoom}
        onZoomChange={setZoom}
      />

      <QRCodeDialog 
        open={showQR}
        onClose={() => setShowQR(false)}
      />
      
      <ShapeDialog 
        open={showShapes}
        onClose={() => setShowShapes(false)}
      />
      
      <KeyboardShortcutsModal 
        open={showHelp}
        onClose={() => setShowHelp(false)}
      />
    </div>
  );
}

export default DesignApp;
