🎨 Digital Art Studio

Advanced drawing and painting application with brushes and layers.

import { useCanvasPan, useCanvasKeyboardShortcuts } from 'fabricjs-design-tool';

function DigitalArtStudio() {
  const [brushSize, setBrushSize] = useState(10);
  const [brushColor, setBrushColor] = useState('#000000');
  
  useCanvasPan(canvas);
  useCanvasKeyboardShortcuts(canvas);
  
  return (
    <div className="art-studio">
      <BrushPanel 
        size={brushSize} 
        color={brushColor}
        onSizeChange={setBrushSize}
        onColorChange={setBrushColor}
      />
      <CanvasWrapper />
      <LayersPanel />
    </div>
  );
}

    this.canvasManager = useCanvasManager(this.canvas);
  }
};
</script>

Angular Integration

// design-tool.component.ts
import { Component, OnInit, ViewChild, ElementRef } from '@angular/core';
import { Canvas } from 'fabric';
import { useCanvasManager } from 'fabricjs-design-tool';

@Component({
  selector: 'app-design-tool',
  template: `
    <canvas #canvas width="800" height="600"></canvas>
  `
})
export class DesignToolComponent implements OnInit {
  @ViewChild('canvas') canvasRef!: ElementRef;
  canvas!: Canvas;
  
  ngOnInit() {
    this.canvas = new Canvas(this.canvasRef.nativeElement);
    const canvasManager = useCanvasManager(this.canvas);
  }
}
